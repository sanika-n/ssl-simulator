333// returns angle in range [-B3_2_PI, B3_2_PI], closest to one of the limits
334// all arguments should be normalized angles (i.e. in range [-B3_PI, B3_PI])
335B3_FORCE_INLINE b3Scalar b3AdjustAngleToLimits(b3Scalar angleInRadians, b3Scalar angleLowerLimitInRadians, b3Scalar angleUpperLimitInRadians)
336{
337 if(angleLowerLimitInRadians >= angleUpperLimitInRadians)
338 {
339 return angleInRadians;
340 }
341 else if(angleInRadians < angleLowerLimitInRadians)
342 {
343 b3Scalar diffLo = b3Fabs(b3NormalizeAngle(angleLowerLimitInRadians - angleInRadians));
344 b3Scalar diffHi = b3Fabs(b3NormalizeAngle(angleUpperLimitInRadians - angleInRadians));
345 return (diffLo < diffHi) ? angleInRadians : (angleInRadians + B3_2_PI);
346 }
347 else if(angleInRadians > angleUpperLimitInRadians)
348 {
349 b3Scalar diffHi = b3Fabs(b3NormalizeAngle(angleInRadians - angleUpperLimitInRadians));
350 b3Scalar diffLo = b3Fabs(b3NormalizeAngle(angleInRadians - angleLowerLimitInRadians));
351 return (diffLo < diffHi) ? (angleInRadians - B3_2_PI) : angleInRadians;
352 }
353 else
354 {
355 return angleInRadians;
356 }
357}
