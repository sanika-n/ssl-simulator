113// stbtt_InitFont()
114// stbtt_GetFontOffsetForIndex() -- use for TTC font collections
115//
116// Render a unicode codepoint to a bitmap
117// stbtt_GetCodepointBitmap() -- allocates and returns a bitmap
118// stbtt_MakeCodepointBitmap() -- renders into bitmap you provide
119// stbtt_GetCodepointBitmapBox() -- how big the bitmap must be
120//
121// Character advance/positioning
122// stbtt_GetCodepointHMetrics()
123// stbtt_GetFontVMetrics()
124// stbtt_GetCodepointKernAdvance()
125//
126// Starting with version 1.06, the rasterizer was replaced with a new,
127// faster and generally-more-precise rasterizer. The new rasterizer more
128// accurately measures pixel coverage for anti-aliasing, except in the case
129// where multiple shapes overlap, in which case it overestimates the AA pixel
130// coverage. Thus, anti-aliasing of intersecting shapes may look wrong. If
131// this turns out to be a problem, you can re-enable the old rasterizer with
132// #define STBTT_RASTERIZER_VERSION 1
133// which will incur about a 15% speed hit.

135// ADDITIONAL DOCUMENTATION
136//
137// Immediately after this block comment are a series of sample programs.
138//
139// After the sample programs is the "header file" section. This section
140// includes documentation for each API function.
141//
142// Some important concepts to understand to use this library:
143//
144// Codepoint
145// Characters are defined by unicode codepoints, e.g. 65 is
146// uppercase A, 231 is lowercase c with a cedilla, 0x7e30 is
147// the hiragana for "ma".
148//
149// Glyph
150// A visual character shape (every codepoint is rendered as
151// some glyph)
152//
153// Glyph index
154// A font-specific integer ID representing a glyph
155//
156// Baseline
157// Glyph shapes are defined relative to a baseline, which is the
158// bottom of uppercase characters. Characters extend both above
159// and below the baseline.
160//
161// Current Point
162// As you draw text to the screen, you keep track of a "current point"
163// which is the origin of each character. The current point's vertical
164// position is the baseline. Even "baked fonts" use this model.
165//
166// Vertical Font Metrics
167// The vertical qualities of the font, used to vertically position
168// and space the characters. See docs for stbtt_GetFontVMetrics.
169//
170// Font Size in Pixels or Points
171// The preferred interface for specifying font sizes in stb_truetype
172// is to specify how tall the font's vertical extent should be in pixels.
173// If that sounds good enough, skip the next paragraph.
174//
175// Most font APIs instead use "points", which are a common typographic
176// measurement for describing font size, defined as 72 points per inch.
177// stb_truetype provides a point API for compatibility. However, true
178// "per inch" conventions don't make much sense on computer displays
179// since they different monitors have different number of pixels per
180// inch. For example, Windows traditionally uses a convention that
181// there are 96 pixels per inch, thus making 'inch' measurements have
182// nothing to do with inches, and thus effectively defining a point to
183// be 1.333 pixels. Additionally, the TrueType font data provides
184// an explicit scale factor to scale a given font's glyphs to points,
185// but the author has observed that this scale factor is often wrong
186// for non-commercial fonts, thus making fonts scaled in points
187// according to the TrueType spec incoherently sized in practice.
188//
189// ADVANCED USAGE
190//
191// Quality:
192//
193// - Use the functions with Subpixel at the end to allow your characters
194// to have subpixel positioning. Since the font is anti-aliased, not
195// hinted, this is very import for quality. (This is not possible with
196// baked fonts.)
197//
198// - Kerning is now supported, and if you're supporting subpixel rendering
199// then kerning is worth using to give your text a polished look.
200//
201// Performance:
202//
203// - Convert Unicode codepoints to glyph indexes and operate on the glyphs;
204// if you don't do this, stb_truetype is forced to do the conversion on
205// every call.
206//
207// - There are a lot of memory allocations. We should modify it to take
208// a temp buffer and allocate from the temp buffer (without freeing),
209// should help performance a lot.
210//
211// NOTES
212//
213// The system uses the raw data found in the .ttf file without changing it
214// and without building auxiliary data structures. This is a bit inefficient
215// on little-endian systems (the data is big-endian), but assuming you're
216// caching the bitmaps or glyph shapes this shouldn't be a big deal.
217//
218// It appears to be very hard to programmatically determine what font a
219// given file is in a general way. I provide an API for this, but I don't
220// recommend it.

966{

970 for (i=0; i < num_tables; ++i) {
971 stbtt_uint32 loc = tabledir + 16*i;

974 }
975 return 0;

997STBTT_DEF int stbtt_InitFont(stbtt_fontinfo *info, const unsigned char *data2, int fontstart)
998{
999 stbtt_uint8 *data = (stbtt_uint8 *) data2;
1000 stbtt_uint32 cmap, t;
1001 stbtt_int32 i,numTables;
1002
1003 info->data = data;
1004 info->fontstart = fontstart;
1005
1006 cmap = stbtt__find_table(data, fontstart, "cmap"); // required
1007 info->loca = stbtt__find_table(data, fontstart, "loca"); // required
1008 info->head = stbtt__find_table(data, fontstart, "head"); // required
1009 info->glyf = stbtt__find_table(data, fontstart, "glyf"); // required
1010 info->hhea = stbtt__find_table(data, fontstart, "hhea"); // required
1011 info->hmtx = stbtt__find_table(data, fontstart, "hmtx"); // required
1012 info->kern = stbtt__find_table(data, fontstart, "kern"); // not required
1013 if (!cmap || !info->loca || !info->head || !info->glyf || !info->hhea || !info->hmtx)
1014 return 0;
1015
1016 t = stbtt__find_table(data, fontstart, "maxp");
1017 if (t)
1018 info->numGlyphs = ttUSHORT(data+t+4);
1019 else
1020 info->numGlyphs = 0xffff;
1021
1022 // find a cmap encoding table we understand *now* to avoid searching
1023 // later. (todo: could make this installable)
1024 // the same regardless of glyph.
1025 numTables = ttUSHORT(data + cmap + 2);
1026 info->index_map = 0;
1027 for (i=0; i < numTables; ++i) {
1028 stbtt_uint32 encoding_record = cmap + 4 + 8 * i;
1029 // find an encoding we understand:
1030 switch(ttUSHORT(data+encoding_record)) {
1031 case STBTT_PLATFORM_ID_MICROSOFT:
1032 switch (ttUSHORT(data+encoding_record+2)) {
1033 case STBTT_MS_EID_UNICODE_BMP:
1034 case STBTT_MS_EID_UNICODE_FULL:
1035 // MS/Unicode
1036 info->index_map = cmap + ttULONG(data+encoding_record+4);
1037 break;
1038 }
1039 break;
1040 case STBTT_PLATFORM_ID_UNICODE:
1041 // Mac/iOS has these
1042 // all the encodingIDs are unicode, so we don't bother to check it
1043 info->index_map = cmap + ttULONG(data+encoding_record+4);
1044 break;
1045 }
1046 }
1047 if (info->index_map == 0)
1048 return 0;
1049
1050 info->indexToLocFormat = ttUSHORT(data+info->head + 50);
1051 return 1;
1052}

1054STBTT_DEF int stbtt_FindGlyphIndex(const stbtt_fontinfo *info, int unicode_codepoint)
1055{
1056 stbtt_uint8 *data = info->data;
1057 stbtt_uint32 index_map = info->index_map;
1058
1059 stbtt_uint16 format = ttUSHORT(data + index_map + 0);
1060 if (format == 0) { // apple byte encoding
1061 stbtt_int32 bytes = ttUSHORT(data + index_map + 2);
1062 if (unicode_codepoint < bytes-6)
1063 return ttBYTE(data + index_map + 6 + unicode_codepoint);
1064 return 0;
1065 } else if (format == 6) {
1066 stbtt_uint32 first = ttUSHORT(data + index_map + 6);
1067 stbtt_uint32 count = ttUSHORT(data + index_map + 8);
1068 if ((stbtt_uint32) unicode_codepoint >= first && (stbtt_uint32) unicode_codepoint < first+count)
1069 return ttUSHORT(data + index_map + 10 + (unicode_codepoint - first)*2);
1070 return 0;
1071 } else if (format == 2) {
1072 STBTT_assert(0); // @TODO: high-byte mapping for japanese/chinese/korean
1073 return 0;
1074 } else if (format == 4) { // standard mapping for windows fonts: binary search collection of ranges
1075 stbtt_uint16 segcount = ttUSHORT(data+index_map+6) >> 1;
1076 stbtt_uint16 searchRange = ttUSHORT(data+index_map+8) >> 1;
1077 stbtt_uint16 entrySelector = ttUSHORT(data+index_map+10);
1078 stbtt_uint16 rangeShift = ttUSHORT(data+index_map+12) >> 1;
1079
1080 // do a binary search of the segments
1081 stbtt_uint32 endCount = index_map + 14;
1082 stbtt_uint32 search = endCount;
1083
1084 if (unicode_codepoint > 0xffff)
1085 return 0;
1086
1087 // they lie from endCount .. endCount + segCount
1088 // but searchRange is the nearest power of two, so...
1089 if (unicode_codepoint >= ttUSHORT(data + search + rangeShift*2))
1090 search += rangeShift*2;
1091
1092 // now decrement to bias correctly to find smallest
1093 search -= 2;
1094 while (entrySelector) {
1095 stbtt_uint16 end;
1096 searchRange >>= 1;
1097 end = ttUSHORT(data + search + searchRange*2);
1098 if (unicode_codepoint > end)
1099 search += searchRange*2;
1100 --entrySelector;
1101 }
1102 search += 2;
1103
1104 {
1105 stbtt_uint16 offset, start;
1106 stbtt_uint16 item = (stbtt_uint16) ((search - endCount) >> 1);
1107
1108 STBTT_assert(unicode_codepoint <= ttUSHORT(data + endCount + 2*item));
1109 start = ttUSHORT(data + index_map + 14 + segcount*2 + 2 + 2*item);
1110 if (unicode_codepoint < start)
1111 return 0;
1112
1113 offset = ttUSHORT(data + index_map + 14 + segcount*6 + 2 + 2*item);
1114 if (offset == 0)
1115 return (stbtt_uint16) (unicode_codepoint + ttSHORT(data + index_map + 14 + segcount*4 + 2 + 2*item));
1116
1117 return ttUSHORT(data + offset + (unicode_codepoint-start)*2 + index_map + 14 + segcount*6 + 2 + 2*item);
1118 }
1119 } else if (format == 12 || format == 13) {
1120 stbtt_uint32 ngroups = ttULONG(data+index_map+12);
1121 stbtt_int32 low,high;
1122 low = 0; high = (stbtt_int32)ngroups;
1123 // Binary search the right group.
1124 while (low < high) {
1125 stbtt_int32 mid = low + ((high-low) >> 1); // rounds down, so low <= mid < high
1126 stbtt_uint32 start_char = ttULONG(data+index_map+16+mid*12);
1127 stbtt_uint32 end_char = ttULONG(data+index_map+16+mid*12+4);
1128 if ((stbtt_uint32) unicode_codepoint < start_char)
1129 high = mid;
1130 else if ((stbtt_uint32) unicode_codepoint > end_char)
1131 low = mid+1;
1132 else {
1133 stbtt_uint32 start_glyph = ttULONG(data+index_map+16+mid*12+8);
1134 if (format == 12)
1135 return start_glyph + unicode_codepoint-start_char;
1136 else // format == 13
1137 return start_glyph;
1138 }
1139 }
1140 return 0; // not found
1141 }
1142 // @TODO
1143 STBTT_assert(0);
1144 return 0;
1145}

1221STBTT_DEF int stbtt_GetGlyphShape(const stbtt_fontinfo *info, int glyph_index, stbtt_vertex **pvertices)
1222{
1223 stbtt_int16 numberOfContours;
1224 stbtt_uint8 *endPtsOfContours;
1225 stbtt_uint8 *data = info->data;
1226 stbtt_vertex *vertices=0;
1227 int num_vertices=0;
1228 int g = stbtt__GetGlyfOffset(info, glyph_index);
1229
1230 *pvertices = NULL;
1231
1232 if (g < 0) return 0;
1233
1234 numberOfContours = ttSHORT(data + g);
1235
1236 if (numberOfContours > 0) {
1237 stbtt_uint8 flags=0,flagcount;
1238 stbtt_int32 ins, i,j=0,m,n, next_move, was_off=0, off, start_off=0;
1239 stbtt_int32 x,y,cx,cy,sx,sy, scx,scy;
1240 stbtt_uint8 *points;
1241 endPtsOfContours = (data + g + 10);
1242 ins = ttUSHORT(data + g + 10 + numberOfContours * 2);
1243 points = data + g + 10 + numberOfContours * 2 + 2 + ins;
1244
1245 n = 1+ttUSHORT(endPtsOfContours + numberOfContours*2-2);
1246
1247 m = n + 2*numberOfContours; // a loose bound on how many vertices we might need
1248 vertices = (stbtt_vertex *) STBTT_malloc(m * sizeof(vertices[0]), info->userdata);
1249 if (vertices == 0)
1250 return 0;
1251
1252 next_move = 0;
1253 flagcount=0;
1254
1255 // in first pass, we load uninterpreted data into the allocated array
1256 // above, shifted to the end of the array so we won't overwrite it when
1257 // we create our final data starting from the front
1258
1259 off = m - n; // starting offset for uninterpreted data, regardless of how m ends up being calculated
1260
1261 // first load flags
1262
1263 for (i=0; i < n; ++i) {
1264 if (flagcount == 0) {
1265 flags = *points++;
1266 if (flags & 8)
1267 flagcount = *points++;
1268 } else
1269 --flagcount;
1270 vertices[off+i].type = flags;
1271 }
1272
1273 // now load x coordinates
1274 x=0;
1275 for (i=0; i < n; ++i) {
1276 flags = vertices[off+i].type;
1277 if (flags & 2) {
1278 stbtt_int16 dx = *points++;
1279 x += (flags & 16) ? dx : -dx; // ???
1280 } else {
1281 if (!(flags & 16)) {
1282 x = x + (stbtt_int16) (points[0]*256 + points[1]);
1283 points += 2;
1284 }
1285 }
1286 vertices[off+i].x = (stbtt_int16) x;
1287 }
1288
1289 // now load y coordinates
1290 y=0;
1291 for (i=0; i < n; ++i) {
1292 flags = vertices[off+i].type;
1293 if (flags & 4) {
1294 stbtt_int16 dy = *points++;
1295 y += (flags & 32) ? dy : -dy; // ???
1296 } else {
1297 if (!(flags & 32)) {
1298 y = y + (stbtt_int16) (points[0]*256 + points[1]);
1299 points += 2;
1300 }
1301 }
1302 vertices[off+i].y = (stbtt_int16) y;
1303 }
1304
1305 // now convert them to our format
1306 num_vertices=0;
1307 sx = sy = cx = cy = scx = scy = 0;
1308 for (i=0; i < n; ++i) {
1309 flags = vertices[off+i].type;
1310 x = (stbtt_int16) vertices[off+i].x;
1311 y = (stbtt_int16) vertices[off+i].y;
1312
1313 if (next_move == i) {
1314 if (i != 0)
1315 num_vertices = stbtt__close_shape(vertices, num_vertices, was_off, start_off, sx,sy,scx,scy,cx,cy);
1316
1317 // now start the new one
1318 start_off = !(flags & 1);
1319 if (start_off) {
1320 // if we start off with an off-curve point, then when we need to find a point on the curve
1321 // where we can start, and we need to save some state for when we wraparound.
1322 scx = x;
1323 scy = y;
1324 if (!(vertices[off+i+1].type & 1)) {
1325 // next point is also a curve point, so interpolate an on-point curve
1326 sx = (x + (stbtt_int32) vertices[off+i+1].x) >> 1;
1327 sy = (y + (stbtt_int32) vertices[off+i+1].y) >> 1;
1328 } else {
1329 // otherwise just use the next point as our start point
1330 sx = (stbtt_int32) vertices[off+i+1].x;
1331 sy = (stbtt_int32) vertices[off+i+1].y;
1332 ++i; // we're using point i+1 as the starting point, so skip it
1333 }
1334 } else {
1335 sx = x;
1336 sy = y;
1337 }
1338 stbtt_setvertex(&vertices[num_vertices++], STBTT_vmove,sx,sy,0,0);
1339 was_off = 0;
1340 next_move = 1 + ttUSHORT(endPtsOfContours+j*2);
1341 ++j;
1342 } else {
1343 if (!(flags & 1)) { // if it's a curve
1344 if (was_off) // two off-curve control points in a row means interpolate an on-curve midpoint
1345 stbtt_setvertex(&vertices[num_vertices++], STBTT_vcurve, (cx+x)>>1, (cy+y)>>1, cx, cy);
1346 cx = x;
1347 cy = y;
1348 was_off = 1;
1349 } else {
1350 if (was_off)
1351 stbtt_setvertex(&vertices[num_vertices++], STBTT_vcurve, x,y, cx, cy);
1352 else
1353 stbtt_setvertex(&vertices[num_vertices++], STBTT_vline, x,y,0,0);
1354 was_off = 0;
1355 }
1356 }
1357 }
1358 num_vertices = stbtt__close_shape(vertices, num_vertices, was_off, start_off, sx,sy,scx,scy,cx,cy);
1359 } else if (numberOfContours == -1) {
1360 // Compound shapes.
1361 int more = 1;
1362 stbtt_uint8 *comp = data + g + 10;
1363 num_vertices = 0;
1364 vertices = 0;
1365 while (more) {
1366 stbtt_uint16 flags, gidx;
1367 int comp_num_verts = 0, i;
1368 stbtt_vertex *comp_verts = 0, *tmp = 0;
1369 float mtx[6] = {1,0,0,1,0,0}, m, n;
1370
1371 flags = ttSHORT(comp); comp+=2;
1372 gidx = ttSHORT(comp); comp+=2;
1373
1374 if (flags & 2) { // XY values
1375 if (flags & 1) { // shorts
1376 mtx[4] = ttSHORT(comp); comp+=2;
1377 mtx[5] = ttSHORT(comp); comp+=2;
1378 } else {
1379 mtx[4] = ttCHAR(comp); comp+=1;
1380 mtx[5] = ttCHAR(comp); comp+=1;
1381 }
1382 }
1383 else {
1384 // @TODO handle matching point
1385 STBTT_assert(0);
1386 }
1387 if (flags & (1<<3)) { // WE_HAVE_A_SCALE
1388 mtx[0] = mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
1389 mtx[1] = mtx[2] = 0;
1390 } else if (flags & (1<<6)) { // WE_HAVE_AN_X_AND_YSCALE
1391 mtx[0] = ttSHORT(comp)/16384.0f; comp+=2;
1392 mtx[1] = mtx[2] = 0;
1393 mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
1394 } else if (flags & (1<<7)) { // WE_HAVE_A_TWO_BY_TWO
1395 mtx[0] = ttSHORT(comp)/16384.0f; comp+=2;
1396 mtx[1] = ttSHORT(comp)/16384.0f; comp+=2;
1397 mtx[2] = ttSHORT(comp)/16384.0f; comp+=2;
1398 mtx[3] = ttSHORT(comp)/16384.0f; comp+=2;
1399 }
1400
1401 // Find transformation scales.
1402 m = (float) STBTT_sqrt(mtx[0]*mtx[0] + mtx[1]*mtx[1]);
1403 n = (float) STBTT_sqrt(mtx[2]*mtx[2] + mtx[3]*mtx[3]);
1404
1405 // Get indexed glyph.
1406 comp_num_verts = stbtt_GetGlyphShape(info, gidx, &comp_verts);
1407 if (comp_num_verts > 0) {
1408 // Transform vertices.
1409 for (i = 0; i < comp_num_verts; ++i) {
1410 stbtt_vertex* v = &comp_verts[i];
1411 stbtt_vertex_type x,y;
1412 x=v->x; y=v->y;
1413 v->x = (stbtt_vertex_type)(m * (mtx[0]*x + mtx[2]*y + mtx[4]));
1414 v->y = (stbtt_vertex_type)(n * (mtx[1]*x + mtx[3]*y + mtx[5]));
1415 x=v->cx; y=v->cy;
1416 v->cx = (stbtt_vertex_type)(m * (mtx[0]*x + mtx[2]*y + mtx[4]));
1417 v->cy = (stbtt_vertex_type)(n * (mtx[1]*x + mtx[3]*y + mtx[5]));
1418 }
1419 // Append vertices.
1420 tmp = (stbtt_vertex*)STBTT_malloc((num_vertices+comp_num_verts)*sizeof(stbtt_vertex), info->userdata);
1421 if (!tmp) {
1422 if (vertices) STBTT_free(vertices, info->userdata);
1423 if (comp_verts) STBTT_free(comp_verts, info->userdata);
1424 return 0;
1425 }
1426 if (num_vertices > 0) STBTT_memcpy(tmp, vertices, num_vertices*sizeof(stbtt_vertex));
1427 STBTT_memcpy(tmp+num_vertices, comp_verts, comp_num_verts*sizeof(stbtt_vertex));
1428 if (vertices) STBTT_free(vertices, info->userdata);
1429 vertices = tmp;
1430 STBTT_free(comp_verts, info->userdata);
1431 num_vertices += comp_num_verts;
1432 }
1433 // More components ?
1434 more = flags & (1<<5);
1435 }
1436 } else if (numberOfContours < 0) {
1437 // @TODO other compound variations?
1438 STBTT_assert(0);
1439 } else {
1440 // numberOfCounters == 0, do nothing
1441 }
1442
1443 *pvertices = vertices;
1444 return num_vertices;
1445}

1731 }
1732 }
1733 }

1736 }

1743 int y,j=0;
1744 int max_weight = (255 / vsubsample); // weight per vertical scanline
1745 int s; // vertical subsample index
1746 unsigned char scanline_data[512], *scanline;

1748 if (result->w > 512)
1749 scanline = (unsigned char *) STBTT_malloc(result->w, userdata);

1759 // find center of pixel for this scanline

1860 assert(x1 >= x);
1861 else if (x0 <= x)
1862 assert(x1 <= x);
1863 else if (x0 >= x+1)
1864 assert(x1 >= x+1);
1865 else

1867
1868 if (x0 <= x && x1 <= x)
1869 scanline[x] += e->direction * (y1-y0);
1870 else if (x0 >= x+1 && x1 >= x+1)
1871 ;
1872 else {
1873 assert(x0 >= x && x0 <= x+1 && x1 >= x && x1 <= x+1);
1874 scanline[x] += e->direction * (y1-y0) * (1-((x0-x)+(x1-x))/2); // coverage = 1 - average x position

1878static void stbtt__fill_active_edges_new(float *scanline, float *scanline_fill, int len, stbtt__active_edge *e, float y_top)
1879{
1880 float y_bottom = y_top+1;
1881
1882 while (e) {
1883 // brute force every pixel
1884
1885 // compute intersection points with top & bottom
1886 assert(e->ey >= y_top);
1887
1888 if (e->fdx == 0) {
1889 float x0 = e->fx;
1890 if (x0 < len) {
1891 if (x0 >= 0) {
1892 stbtt__handle_clipped_edge(scanline,(int) x0,e, x0,y_top, x0,y_bottom);
1893 stbtt__handle_clipped_edge(scanline_fill-1,(int) x0+1,e, x0,y_top, x0,y_bottom);
1894 } else {
1895 stbtt__handle_clipped_edge(scanline_fill-1,0,e, x0,y_top, x0,y_bottom);
1896 }
1897 }
1898 } else {
1899 float x0 = e->fx;
1900 float dx = e->fdx;
1901 float xb = x0 + dx;
1902 float x_top, x_bottom;
1903 float y0,y1;
1904 float dy = e->fdy;
1905 assert(e->sy <= y_bottom && e->ey >= y_top);
1906
1907 // compute endpoints of line segment clipped to this scanline (if the
1908 // line segment starts on this scanline. x0 is the intersection of the
1909 // line with y_top, but that may be off the line segment.
1910 if (e->sy > y_top) {
1911 x_top = x0 + dx * (e->sy - y_top);
1912 y0 = e->sy;
1913 } else {
1914 x_top = x0;
1915 y0 = y_top;
1916 }
1917 if (e->ey < y_bottom) {
1918 x_bottom = x0 + dx * (e->ey - y_top);
1919 y1 = e->ey;
1920 } else {
1921 x_bottom = xb;
1922 y1 = y_bottom;
1923 }
1924
1925 if (x_top >= 0 && x_bottom >= 0 && x_top < len && x_bottom < len) {
1926 // from here on, we don't have to range check x values
1927
1928 if ((int) x_top == (int) x_bottom) {
1929 float height;
1930 // simple case, only spans one pixel
1931 int x = (int) x_top;
1932 height = y1 - y0;
1933 assert(x >= 0 && x < len);
1934 scanline[x] += e->direction * (1-((x_top - x) + (x_bottom-x))/2) * height;
1935 scanline_fill[x] += e->direction * height; // everything right of this pixel is filled
1936 } else {
1937 int x,x1,x2;
1938 float y_crossing, step, sign, area;
1939 // covers 2+ pixels
1940 if (x_top > x_bottom) {
1941 // flip scanline vertically; signed area is the same
1942 float t;
1943 y0 = y_bottom - (y0 - y_top);
1944 y1 = y_bottom - (y1 - y_top);
1945 t = y0, y0 = y1, y1 = t;
1946 t = x_bottom, x_bottom = x_top, x_top = t;
1947 dx = -dx;
1948 dy = -dy;
1949 t = x0, x0 = xb, xb = t;
1950 }
1951
1952 x1 = (int) x_top;
1953 x2 = (int) x_bottom;
1954 // compute intersection with y axis at x1+1
1955 y_crossing = (x1+1 - x0) * dy + y_top;
1956
1957 sign = e->direction;
1958 // area of the rectangle covered from y0..y_crossing
1959 area = sign * (y_crossing-y0);
1960 // area of the triangle (x_top,y0), (x+1,y0), (x+1,y_crossing)
1961 scanline[x1] += area * (1-((x_top - x1)+(x1+1-x1))/2);
1962
1963 step = sign * dy;
1964 for (x = x1+1; x < x2; ++x) {
1965 scanline[x] += area + step/2;
1966 area += step;
1967 }
1968 y_crossing += dy * (x2 - (x1+1));
1969
1970 assert(fabs(area) <= 1.01f);
1971
1972 scanline[x2] += area + sign * (1-((x2-x2)+(x_bottom-x2))/2) * (y1-y_crossing);
1973
1974 scanline_fill[x2] += sign * (y1-y0);
1975 }
1976 } else {
1977 // if edge goes outside of box we're drawing, we require
1978 // clipping logic. since this does not match the intended use
1979 // of this library, we use a different, very slow brute
1980 // force implementation
1981 int x;
1982 for (x=0; x < len; ++x) {
1983 // cases:
1984 //
1985 // there can be up to two intersections with the pixel. any intersection
1986 // with left or right edges can be handled by splitting into two (or three)
1987 // regions. intersections with top & bottom do not necessitate case-wise logic.
1988 float y0,y1;
1989 float y_cur = y_top, x_cur = x0;
1990 // x = e->x + e->dx * (y-y_top)
1991 // (y-y_top) = (x - e->x) / e->dx
1992 // y = (x - e->x) / e->dx + y_top
1993 y0 = (x - x0) / dx + y_top;
1994 y1 = (x+1 - x0) / dx + y_top;
1995
1996 if (y0 < y1) {
1997 if (y0 > y_top && y0 < y_bottom) {
1998 stbtt__handle_clipped_edge(scanline,x,e, x_cur,y_cur, (float) x,y0);
1999 y_cur = y0;
2000 x_cur = (float) x;
2001 }
2002 if (y1 >= y_cur && y1 < y_bottom) {
2003 stbtt__handle_clipped_edge(scanline,x,e, x_cur,y_cur, (float) x+1,y1);
2004 y_cur = y1;
2005 x_cur = (float) x+1;
2006 }
2007 } else {
2008 if (y1 >= y_cur && y1 < y_bottom) {
2009 stbtt__handle_clipped_edge(scanline,x,e, x_cur,y_cur, (float) x+1,y1);
2010 y_cur = y1;
2011 x_cur = (float) x+1;
2012 }
2013 if (y0 > y_top && y0 < y_bottom) {
2014 stbtt__handle_clipped_edge(scanline,x,e, x_cur,y_cur, (float) x,y0);
2015 y_cur = y0;
2016 x_cur = (float) x;
2017 }
2018 }
2019 stbtt__handle_clipped_edge(scanline,x,e, x_cur,y_cur, xb,y_bottom);
2020 }
2021 }
2022 }
2023 e = e->next;
2024 }

2136static void stbtt__sort_edges_quicksort(stbtt__edge *p, int n)
2137{
2138 /* threshhold for transitioning to insertion sort */
2139 while (n > 12) {
2140 stbtt__edge t;
2141 int c01,c12,c,m,i,j;
2142
2143 /* compute median of three */
2144 m = n >> 1;
2145 c01 = STBTT__COMPARE(&p[0],&p[m]);
2146 c12 = STBTT__COMPARE(&p[m],&p[n-1]);
2147 /* if 0 >= mid >= end, or 0 < mid < end, then use mid */
2148 if (c01 != c12) {
2149 /* otherwise, we'll need to swap something else to middle */
2150 int z;
2151 c = STBTT__COMPARE(&p[0],&p[n-1]);
2152 /* 0>mid && mid<n: 0>n => n; 0<n => 0 */
2153 /* 0<mid && mid>n: 0>n => 0; 0<n => n */
2154 z = (c == c12) ? 0 : n-1;
2155 t = p[z];
2156 p[z] = p[m];
2157 p[m] = t;
2158 }
2159 /* now p[m] is the median-of-three */
2160 /* swap it to the beginning so it won't move around */
2161 t = p[0];
2162 p[0] = p[m];
2163 p[m] = t;
2164
2165 /* partition loop */
2166 i=1;
2167 j=n-1;
2168 for(;;) {
2169 /* handling of equality is crucial here */
2170 /* for sentinels & efficiency with duplicates */
2171 for (;;++i) {
2172 if (!STBTT__COMPARE(&p[i], &p[0])) break;
2173 }
2174 for (;;--j) {
2175 if (!STBTT__COMPARE(&p[0], &p[j])) break;
2176 }
2177 /* make sure we haven't crossed */
2178 if (i >= j) break;
2179 t = p[i];
2180 p[i] = p[j];
2181 p[j] = t;
2182
2183 ++i;
2184 --j;
2185 }
2186 /* recurse on smaller side, iterate on larger */
2187 if (j < (n-i)) {
2188 stbtt__sort_edges_quicksort(p,j);
2189 p = p+i;
