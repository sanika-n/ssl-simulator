23// Clips a face to the back of a plane, return the number of vertices out, stored in ppVtxOut
24inline int b3ClipFace(const b3Float4* pVtxIn, int numVertsIn, b3Float4& planeNormalWS,float planeEqWS, b3Float4* ppVtxOut)
25{
26
27 int ve;
28 float ds, de;
29 int numVertsOut = 0;
30 if (numVertsIn < 2)
31 return 0;
32
33 b3Float4 firstVertex=pVtxIn[numVertsIn-1];
34 b3Float4 endVertex = pVtxIn[0];
35
36 ds = b3Dot3F4(planeNormalWS,firstVertex)+planeEqWS;
37
38 for (ve = 0; ve < numVertsIn; ve++)
39 {
40 endVertex=pVtxIn[ve];
41
42 de = b3Dot3F4(planeNormalWS,endVertex)+planeEqWS;
43
44 if (ds<0)
45 {
46 if (de<0)
47 {
48 // Start < 0, end < 0, so output endVertex
49 ppVtxOut[numVertsOut++] = endVertex;
50 }
51 else
52 {
53 // Start < 0, end >= 0, so output intersection
54 ppVtxOut[numVertsOut++] = b3Lerp3(firstVertex, endVertex,(ds * 1.f/(ds - de)) );
55 }
56 }
57 else
58 {
59 if (de<0)
60 {
61 // Start >= 0, end < 0 so output intersection and end
62 ppVtxOut[numVertsOut++] = b3Lerp3(firstVertex, endVertex,(ds * 1.f/(ds - de)) );
63 ppVtxOut[numVertsOut++] = endVertex;
64 }
65 }
66 firstVertex = endVertex;
67 ds = de;
68 }
69 return numVertsOut;
70}
