43 static void integrateTransform(const b3Transform& curTrans,const b3Vector3& linvel,const b3Vector3& angvel,b3Scalar timeStep,b3Transform& predictedTransform)
44 {
45 predictedTransform.setOrigin(curTrans.getOrigin() + linvel * timeStep);
46// #define QUATERNION_DERIVATIVE
47 #ifdef QUATERNION_DERIVATIVE
48 b3Quaternion predictedOrn = curTrans.getRotation();
49 predictedOrn += (angvel * predictedOrn) * (timeStep * b3Scalar(0.5));
50 predictedOrn.normalize();
51 #else
52 //Exponential map
53 //google for "Practical Parameterization of Rotations Using the Exponential Map", F. Sebastian Grassia
54
55 b3Vector3 axis;
56 b3Scalar fAngle = angvel.length();
57 //limit the angular motion
58 if (fAngle*timeStep > B3_ANGULAR_MOTION_THRESHOLD)
59 {
60 fAngle = B3_ANGULAR_MOTION_THRESHOLD / timeStep;
61 }
62
63 if ( fAngle < b3Scalar(0.001) )
64 {
65 // use Taylor's expansions of sync function
66 axis = angvel*( b3Scalar(0.5)*timeStep-(timeStep*timeStep*timeStep)*(b3Scalar(0.020833333333))*fAngle*fAngle );
67 }
68 else
69 {
70 // sync(fAngle) = sin(c*fAngle)/t
71 axis = angvel*( b3Sin(b3Scalar(0.5)*fAngle*timeStep)/fAngle );
72 }
73 b3Quaternion dorn (axis.getX(),axis.getY(),axis.getZ(),b3Cos( fAngle*timeStep*b3Scalar(0.5) ));
74 b3Quaternion orn0 = curTrans.getRotation();
75
76 b3Quaternion predictedOrn = dorn * orn0;
77 predictedOrn.normalize();
78 #endif
79 predictedTransform.setRotation(predictedOrn);
80 }

97 static void calculateDiffAxisAngleQuaternion(const b3Quaternion& orn0,const b3Quaternion& orn1a,b3Vector3& axis,b3Scalar& angle)
98 {
99 b3Quaternion orn1 = orn0.nearest(orn1a);
100 b3Quaternion dorn = orn1 * orn0.inverse();
101 angle = dorn.getAngle();
102 axis = b3MakeVector3(dorn.getX(),dorn.getY(),dorn.getZ());
103 axis[3] = b3Scalar(0.);
104 //check for axis length
105 b3Scalar len = axis.length2();
106 if (len < B3_EPSILON*B3_EPSILON)
107 axis = b3MakeVector3(b3Scalar(1.),b3Scalar(0.),b3Scalar(0.));
108 else
109 axis /= b3Sqrt(len);
110 }

114 linVel = (transform1.getOrigin() - transform0.getOrigin()) / timeStep;

121 static void calculateDiffAxisAngle(const b3Transform& transform0,const b3Transform& transform1,b3Vector3& axis,b3Scalar& angle)
122 {
123 b3Matrix3x3 dmat = transform1.getBasis() * transform0.getBasis().inverse();
124 b3Quaternion dorn;
125 dmat.getRotation(dorn);
126
128 dorn.normalize();
129
130 angle = dorn.getAngle();
131 axis = b3MakeVector3(dorn.getX(),dorn.getY(),dorn.getZ());
132 axis[3] = b3Scalar(0.);

173 void updateSeparatingDistance(const b3Transform& transA,const b3Transform& transB)
174 {
175 const b3Vector3& toPosA = transA.getOrigin();
176 const b3Vector3& toPosB = transB.getOrigin();
177 b3Quaternion toOrnA = transA.getRotation();
178 b3Quaternion toOrnB = transB.getRotation();
179
180 if (m_separatingDistance>0.f)
181 {
182
183
184 b3Vector3 linVelA,angVelA,linVelB,angVelB;
185 b3TransformUtil::calculateVelocityQuaternion(m_posA,toPosA,m_ornA,toOrnA,b3Scalar(1.),linVelA,angVelA);
186 b3TransformUtil::calculateVelocityQuaternion(m_posB,toPosB,m_ornB,toOrnB,b3Scalar(1.),linVelB,angVelB);
187 b3Scalar maxAngularProjectedVelocity = angVelA.length() * m_boundingRadiusA + angVelB.length() * m_boundingRadiusB;
188 b3Vector3 relLinVel = (linVelB-linVelA);
189 b3Scalar relLinVelocLength = relLinVel.dot(m_separatingNormal);
190 if (relLinVelocLength<0.f)
191 {
192 relLinVelocLength = 0.f;
193 }
194
195 b3Scalar projectedMotion = maxAngularProjectedVelocity +relLinVelocLength;
196 m_separatingDistance -= projectedMotion;
197 }
198
199 m_posA = toPosA;
200 m_posB = toPosB;
201 m_ornA = toOrnA;
202 m_ornB = toOrnB;
203 }

217 m_posA = toPosA;
