177 btVector3 getRayTo(int x,int y)
178 {
179 CommonRenderInterface* renderer = m_guiHelper->getRenderInterface();
180
181 if (!renderer)
182 {
183 btAssert(0);
184 return btVector3(0,0,0);
185 }
186
187 float top = 1.f;
188 float bottom = -1.f;
189 float nearPlane = 1.f;
190 float tanFov = (top-bottom)*0.5f / nearPlane;
191 float fov = btScalar(2.0) * btAtan(tanFov);
192
193 btVector3 camPos,camTarget;
194
195 renderer->getActiveCamera()->getCameraPosition(camPos);
196 renderer->getActiveCamera()->getCameraTargetPosition(camTarget);
197
198 btVector3 rayFrom = camPos;
199 btVector3 rayForward = (camTarget-camPos);
200 rayForward.normalize();
201 float farPlane = 10000.f;
202 rayForward*= farPlane;
203
204 btVector3 rightOffset;
205 btVector3 cameraUp=btVector3(0,0,0);
206 cameraUp[m_guiHelper->getAppInterface()->getUpAxis()]=1;
207
208 btVector3 vertical = cameraUp;
209
210 btVector3 hor;
211 hor = rayForward.cross(vertical);
212 hor.safeNormalize();
213 vertical = hor.cross(rayForward);
214 vertical.safeNormalize();
215
216 float tanfov = tanf(0.5f*fov);
217
218
219 hor *= 2.f * farPlane * tanfov;
220 vertical *= 2.f * farPlane * tanfov;
221
222 btScalar aspect;
223 float width = float(renderer->getScreenWidth());
224 float height = float (renderer->getScreenHeight());
225
226 aspect = width / height;
227
228 hor*=aspect;
229
230
231 btVector3 rayToCenter = rayFrom + rayForward;
232 btVector3 dHor = hor * 1.f/width;
233 btVector3 dVert = vertical * 1.f/height;
234
235
236 btVector3 rayTo = rayToCenter - 0.5f * hor + 0.5f * vertical;
237 rayTo += btScalar(x) * dHor;
238 rayTo -= btScalar(y) * dVert;
239 return rayTo;
240 }

414 {
415 btAssert((!shape || shape->getShapeType() != INVALID_SHAPE_PROXYTYPE));
416

419

423

425
426#define USE_MOTIONSTATE 1
427#ifdef USE_MOTIONSTATE

429

431
432 btRigidBody* body = new btRigidBody(cInfo);
433 //body->setContactProcessingThreshold(m_defaultContactProcessingThreshold);
434
435#else
436 btRigidBody* body = new btRigidBody(mass, 0, shape, localInertia);
437 body->setWorldTransform(startTransform);
438#endif//

440 body->setUserIndex(-1);
441 m_dynamicsWorld->addRigidBody(body);
