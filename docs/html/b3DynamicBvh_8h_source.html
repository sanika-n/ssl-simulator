483 case (0+0+4): px=b3MakeVector3(mi.x,mi.y,mx.z);

485 case (1+0+4): px=b3MakeVector3(mx.x,mi.y,mx.z);
486 pi=b3MakeVector3(mi.x,mx.y,mi.z);break;
487 case (0+2+4): px=b3MakeVector3(mi.x,mx.y,mx.z);

605 __m128 t0(_mm_movehl_ps(ami,ami));

608 __m128 t1(_mm_movehl_ps(bmi,bmi));
