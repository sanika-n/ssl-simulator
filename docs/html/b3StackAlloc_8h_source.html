34class b3StackAlloc
35{
36public:

38 b3StackAlloc(unsigned int size) { ctor();create(size); }

41 inline void create(unsigned int size)
42 {
43 destroy();
44 data = (unsigned char*) b3AlignedAlloc(size,16);
45 totalsize = size;
46 }
47 inline void destroy()
48 {
49 b3Assert(usedsize==0);
50 //Raise(L"StackAlloc is still in use");
51
52 if(usedsize==0)
53 {
54 if(!ischild && data)
55 b3AlignedFree(data);
56
57 data = 0;
58 usedsize = 0;
59 }
60
61 }

63 int getAvailableMemory() const
64 {
65 return static_cast<int>(totalsize - usedsize);
66 }
67
68 unsigned char* allocate(unsigned int size)
69 {
70 const unsigned int nus(usedsize+size);
71 if(nus<totalsize)
72 {
73 usedsize=nus;
74 return(data+(usedsize-size));
75 }
76 b3Assert(0);
77 //&& (L"Not enough memory"));
78
79 return(0);
80 }
81 B3_FORCE_INLINE b3Block* beginBlock()
82 {
83 b3Block* pb = (b3Block*)allocate(sizeof(b3Block));
84 pb->previous = current;
85 pb->address = data+usedsize;
86 current = pb;
87 return(pb);
88 }
89 B3_FORCE_INLINE void endBlock(b3Block* block)
90 {
91 b3Assert(block==current);
92 //Raise(L"Unmatched blocks");
93 if(block==current)
94 {
95 current = block->previous;
96 usedsize = (unsigned int)((block->address-data)-sizeof(b3Block));
97 }
98 }

100private:
101 void ctor()
102 {
103 data = 0;
104 totalsize = 0;
105 usedsize = 0;
106 current = 0;
107 ischild = false;
108 }
109 unsigned char* data;
