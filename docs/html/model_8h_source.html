116 void initTree(std::map<std::string, std::string> &parent_link_tree)
117 {
118 // loop through all joints, for every link, assign children links and children joints
119 for (std::map<std::string,my_shared_ptr<Joint> >::iterator joint = this->joints_.begin();joint != this->joints_.end(); joint++)
120 {
121 std::string parent_link_name = joint->second->parent_link_name;
122 std::string child_link_name = joint->second->child_link_name;
123
124 if (parent_link_name.empty() || child_link_name.empty())
125 {
126 assert(0);
127
128 // throw ParseError("Joint [" + joint->second->name + "] is missing a parent and/or child link specification.");
129 }
130 else
131 {
132 // find child and parent links
133 my_shared_ptr<Link> child_link, parent_link;
134 this->getLink(child_link_name, child_link);
135 if (!child_link)
136 {
137 printf("Error: child link [%s] of joint [%s] not found\n", child_link_name.c_str(),joint->first.c_str() );
138 assert(0);
139// throw ParseError("child link [" + child_link_name + "] of joint [" + joint->first + "] not found");
140 }
141 this->getLink(parent_link_name, parent_link);
142 if (!parent_link)
143 {
144 assert(0);
145
146/* throw ParseError("parent link [" + parent_link_name + "] of joint [" + joint->first + "] not found. This is not valid according to the URDF spec. Every link you refer to from a joint needs to be explicitly defined in the robot description. To fix this problem you can either remove this joint [" + joint->first + "] from your urdf file, or add \"<link name=\"" + parent_link_name + "\" />\" to your urdf file.");
147
148 */}
149
150 //set parent link for child link
151 child_link->setParent(parent_link);
152
153 //set parent joint for child link
154 child_link->parent_joint = joint->second;
155
156 //set child joint for parent link
157 parent_link->child_joints.push_back(joint->second);
158
159 //set child link for parent link
160 parent_link->child_links.push_back(child_link);
161
162 // fill in child/parent string map
163 parent_link_tree[child_link->name] = parent_link_name;
164 }
165 }

168 void initRoot(const std::map<std::string, std::string> &parent_link_tree)
169 {
170 this->root_link_.reset(0);
171
172 // find the links that have no parent in the tree
173 for (std::map<std::string, my_shared_ptr<Link> >::const_iterator l=this->links_.begin(); l!=this->links_.end(); l++)
174 {
175 std::map<std::string, std::string >::const_iterator parent = parent_link_tree.find(l->first);
176 if (parent == parent_link_tree.end())
177 {
178 // store root link
179 if (!this->root_link_)
180 {
181 getLink(l->first, this->root_link_);
182 }
183 // we already found a root link
184 else
185 {
186 assert(0);
187 // throw ParseError("Two root links found: [" + this->root_link_->name + "] and [" + l->first + "]");
188 }
189 }
190 }
191 if (!this->root_link_)
192 {
193 assert(0);
194 //throw ParseError("No root link found. The robot xml is not a valid tree.");
195 }
196 }
