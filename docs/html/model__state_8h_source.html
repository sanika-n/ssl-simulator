52class Time
53{
54public:
55 Time() { this->clear(); };
56
57 void set(double _seconds)
58 {
59 this->sec = (int32_t)(floor(_seconds));
60 this->nsec = (int32_t)(round((_seconds - this->sec) * 1e9));
61 this->Correct();
62 };
63
64 operator double ()
65 {
66 return (static_cast<double>(this->sec) +
67 static_cast<double>(this->nsec)*1e-9);
68 };
69
70 int32_t sec;
71 int32_t nsec;
72
73 void clear()
74 {
75 this->sec = 0;
76 this->nsec = 0;
77 };
78private:
79 void Correct()
80 {
81 // Make any corrections
82 if (this->nsec >= 1e9)
83 {
84 this->sec++;
85 this->nsec = (int32_t)(this->nsec - 1e9);
86 }
87 else if (this->nsec < 0)
88 {
89 this->sec--;
90 this->nsec = (int32_t)(this->nsec + 1e9);
91 }
92 };
93};
