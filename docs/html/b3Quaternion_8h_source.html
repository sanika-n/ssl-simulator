113 void setEuler(const b3Scalar& yaw, const b3Scalar& pitch, const b3Scalar& roll)
114 {
115 b3Scalar halfYaw = b3Scalar(yaw) * b3Scalar(0.5);
116 b3Scalar halfPitch = b3Scalar(pitch) * b3Scalar(0.5);
117 b3Scalar halfRoll = b3Scalar(roll) * b3Scalar(0.5);
118 b3Scalar cosYaw = b3Cos(halfYaw);
119 b3Scalar sinYaw = b3Sin(halfYaw);
120 b3Scalar cosPitch = b3Cos(halfPitch);
121 b3Scalar sinPitch = b3Sin(halfPitch);
122 b3Scalar cosRoll = b3Cos(halfRoll);
123 b3Scalar sinRoll = b3Sin(halfRoll);
124 setValue(cosRoll * sinPitch * cosYaw + sinRoll * cosPitch * sinYaw,
125 cosRoll * cosPitch * sinYaw - sinRoll * sinPitch * cosYaw,
126 sinRoll * cosPitch * cosYaw - cosRoll * sinPitch * sinYaw,
127 cosRoll * cosPitch * cosYaw + sinRoll * sinPitch * sinYaw);
128 }
133 void setEulerZYX(const b3Scalar& yaw, const b3Scalar& pitch, const b3Scalar& roll)
134 {
135 b3Scalar halfYaw = b3Scalar(yaw) * b3Scalar(0.5);
136 b3Scalar halfPitch = b3Scalar(pitch) * b3Scalar(0.5);
137 b3Scalar halfRoll = b3Scalar(roll) * b3Scalar(0.5);
138 b3Scalar cosYaw = b3Cos(halfYaw);
139 b3Scalar sinYaw = b3Sin(halfYaw);
140 b3Scalar cosPitch = b3Cos(halfPitch);
141 b3Scalar sinPitch = b3Sin(halfPitch);
142 b3Scalar cosRoll = b3Cos(halfRoll);
143 b3Scalar sinRoll = b3Sin(halfRoll);
144 setValue(sinRoll * cosPitch * cosYaw - cosRoll * sinPitch * sinYaw, //x
145 cosRoll * sinPitch * cosYaw + sinRoll * cosPitch * sinYaw, //y
146 cosRoll * cosPitch * sinYaw - sinRoll * sinPitch * cosYaw, //z
147 cosRoll * cosPitch * cosYaw + sinRoll * sinPitch * sinYaw); //formerly yzx
148 }

205 b3Quaternion& operator*=(const b3Quaternion& q)
206 {
207#if defined (B3_USE_SSE_IN_API) && defined (B3_USE_SSE)
208 __m128 vQ2 = q.get128();
209
210 __m128 A1 = b3_pshufd_ps(mVec128, B3_SHUFFLE(0,1,2,0));
211 __m128 B1 = b3_pshufd_ps(vQ2, B3_SHUFFLE(3,3,3,0));
212
213 A1 = A1 * B1;
214
215 __m128 A2 = b3_pshufd_ps(mVec128, B3_SHUFFLE(1,2,0,1));
216 __m128 B2 = b3_pshufd_ps(vQ2, B3_SHUFFLE(2,0,1,1));
217
218 A2 = A2 * B2;
219
220 B1 = b3_pshufd_ps(mVec128, B3_SHUFFLE(2,0,1,2));
221 B2 = b3_pshufd_ps(vQ2, B3_SHUFFLE(1,2,0,2));
222
223 B1 = B1 * B2; // A3 *= B3
224
225 mVec128 = b3_splat_ps(mVec128, 3); // A0
226 mVec128 = mVec128 * vQ2; // A0 * B0
227
228 A1 = A1 + A2; // AB12
229 mVec128 = mVec128 - B1; // AB03 = AB0 - AB3
230 A1 = _mm_xor_ps(A1, b3vPPPM); // change sign of the last element
231 mVec128 = mVec128+ A1; // AB03 + AB12
232
233#elif defined(B3_USE_NEON)
234
235 float32x4_t vQ1 = mVec128;
236 float32x4_t vQ2 = q.get128();
237 float32x4_t A0, A1, B1, A2, B2, A3, B3;
238 float32x2_t vQ1zx, vQ2wx, vQ1yz, vQ2zx, vQ2yz, vQ2xz;
239
240 {
241 float32x2x2_t tmp;
242 tmp = vtrn_f32( vget_high_f32(vQ1), vget_low_f32(vQ1) ); // {z x}, {w y}
243 vQ1zx = tmp.val[0];
244
245 tmp = vtrn_f32( vget_high_f32(vQ2), vget_low_f32(vQ2) ); // {z x}, {w y}
246 vQ2zx = tmp.val[0];
247 }
248 vQ2wx = vext_f32(vget_high_f32(vQ2), vget_low_f32(vQ2), 1);
249
250 vQ1yz = vext_f32(vget_low_f32(vQ1), vget_high_f32(vQ1), 1);
251
252 vQ2yz = vext_f32(vget_low_f32(vQ2), vget_high_f32(vQ2), 1);
253 vQ2xz = vext_f32(vQ2zx, vQ2zx, 1);
254
255 A1 = vcombine_f32(vget_low_f32(vQ1), vQ1zx); // X Y z x
256 B1 = vcombine_f32(vdup_lane_f32(vget_high_f32(vQ2), 1), vQ2wx); // W W W X
257
258 A2 = vcombine_f32(vQ1yz, vget_low_f32(vQ1));
259 B2 = vcombine_f32(vQ2zx, vdup_lane_f32(vget_low_f32(vQ2), 1));
260
261 A3 = vcombine_f32(vQ1zx, vQ1yz); // Z X Y Z
262 B3 = vcombine_f32(vQ2yz, vQ2xz); // Y Z x z
263
264 A1 = vmulq_f32(A1, B1);
265 A2 = vmulq_f32(A2, B2);
266 A3 = vmulq_f32(A3, B3); // A3 *= B3
267 A0 = vmulq_lane_f32(vQ2, vget_high_f32(vQ1), 1); // A0 * B0
268
269 A1 = vaddq_f32(A1, A2); // AB12 = AB1 + AB2
270 A0 = vsubq_f32(A0, A3); // AB03 = AB0 - AB3
271
272 // change the sign of the last element
273 A1 = (b3SimdFloat4)veorq_s32((int32x4_t)A1, (int32x4_t)b3vPPPM);
274 A0 = vaddq_f32(A0, A1); // AB03 + AB12
275
276 mVec128 = A0;
277#else
278 setValue(
279 m_floats[3] * q.getX() + m_floats[0] * q.m_floats[3] + m_floats[1] * q.getZ() - m_floats[2] * q.getY(),
280 m_floats[3] * q.getY() + m_floats[1] * q.m_floats[3] + m_floats[2] * q.getX() - m_floats[0] * q.getZ(),
281 m_floats[3] * q.getZ() + m_floats[2] * q.m_floats[3] + m_floats[0] * q.getY() - m_floats[1] * q.getX(),
282 m_floats[3] * q.m_floats[3] - m_floats[0] * q.getX() - m_floats[1] * q.getY() - m_floats[2] * q.getZ());
283#endif
284 return *this;
285 }
288 b3Scalar dot(const b3Quaternion& q) const
289 {
290#if defined (B3_USE_SSE_IN_API) && defined (B3_USE_SSE)
291 __m128 vd;
292
293 vd = _mm_mul_ps(mVec128, q.mVec128);
294
295 __m128 t = _mm_movehl_ps(vd, vd);
296 vd = _mm_add_ps(vd, t);
297 t = _mm_shuffle_ps(vd, vd, 0x55);
298 vd = _mm_add_ss(vd, t);
299
300 return _mm_cvtss_f32(vd);
301#elif defined(B3_USE_NEON)
302 float32x4_t vd = vmulq_f32(mVec128, q.mVec128);
303 float32x2_t x = vpadd_f32(vget_low_f32(vd), vget_high_f32(vd));
304 x = vpadd_f32(x, x);
305 return vget_lane_f32(x, 0);
306#else
307 return m_floats[0] * q.getX() +
308 m_floats[1] * q.getY() +
309 m_floats[2] * q.getZ() +
310 m_floats[3] * q.m_floats[3];
311#endif
312 }

328 b3Quaternion& normalize()
329 {
330#if defined (B3_USE_SSE_IN_API) && defined (B3_USE_SSE)
331 __m128 vd;
332
333 vd = _mm_mul_ps(mVec128, mVec128);
334
335 __m128 t = _mm_movehl_ps(vd, vd);
336 vd = _mm_add_ps(vd, t);
337 t = _mm_shuffle_ps(vd, vd, 0x55);
338 vd = _mm_add_ss(vd, t);
339
340 vd = _mm_sqrt_ss(vd);
341 vd = _mm_div_ss(b3vOnes, vd);
342 vd = b3_pshufd_ps(vd, 0); // splat
343 mVec128 = _mm_mul_ps(mVec128, vd);
344
345 return *this;
346#else
347 return *this /= length();
348#endif

497 b3Quaternion slerp(const b3Quaternion& q, const b3Scalar& t) const
498 {
499 b3Scalar magnitude = b3Sqrt(length2() * q.length2());
500 b3Assert(magnitude > b3Scalar(0));
501
502 b3Scalar product = dot(q) / magnitude;
503 if (b3Fabs(product) < b3Scalar(1))
504 {
505 // Take care of long angle case see http://en.wikipedia.org/wiki/Slerp
506 const b3Scalar sign = (product < 0) ? b3Scalar(-1) : b3Scalar(1);
507
508 const b3Scalar theta = b3Acos(sign * product);
509 const b3Scalar s1 = b3Sin(sign * t * theta);
510 const b3Scalar d = b3Scalar(1.0) / b3Sin(theta);
511 const b3Scalar s0 = b3Sin((b3Scalar(1.0) - t) * theta);
512
513 return b3Quaternion(
514 (m_floats[0] * s0 + q.getX() * s1) * d,
515 (m_floats[1] * s0 + q.getY() * s1) * d,
516 (m_floats[2] * s0 + q.getZ() * s1) * d,
517 (m_floats[3] * s0 + q.m_floats[3] * s1) * d);
518 }
519 else
520 {
521 return *this;
522 }
523 }

541B3_FORCE_INLINE b3Quaternion
542operator*(const b3Quaternion& q1, const b3Quaternion& q2)
543{
544#if defined (B3_USE_SSE_IN_API) && defined (B3_USE_SSE)
545 __m128 vQ1 = q1.get128();
546 __m128 vQ2 = q2.get128();
547 __m128 A0, A1, B1, A2, B2;
548
549 A1 = b3_pshufd_ps(vQ1, B3_SHUFFLE(0,1,2,0)); // X Y z x // vtrn
550 B1 = b3_pshufd_ps(vQ2, B3_SHUFFLE(3,3,3,0)); // W W W X // vdup vext
551
552 A1 = A1 * B1;
553
554 A2 = b3_pshufd_ps(vQ1, B3_SHUFFLE(1,2,0,1)); // Y Z X Y // vext
555 B2 = b3_pshufd_ps(vQ2, B3_SHUFFLE(2,0,1,1)); // z x Y Y // vtrn vdup
556
557 A2 = A2 * B2;
558
559 B1 = b3_pshufd_ps(vQ1, B3_SHUFFLE(2,0,1,2)); // z x Y Z // vtrn vext
560 B2 = b3_pshufd_ps(vQ2, B3_SHUFFLE(1,2,0,2)); // Y Z x z // vext vtrn
561
562 B1 = B1 * B2; // A3 *= B3
563
564 A0 = b3_splat_ps(vQ1, 3); // A0
565 A0 = A0 * vQ2; // A0 * B0
566
567 A1 = A1 + A2; // AB12
568 A0 = A0 - B1; // AB03 = AB0 - AB3
569
570 A1 = _mm_xor_ps(A1, b3vPPPM); // change sign of the last element
571 A0 = A0 + A1; // AB03 + AB12
572
573 return b3Quaternion(A0);
574
575#elif defined(B3_USE_NEON)
576
577 float32x4_t vQ1 = q1.get128();
578 float32x4_t vQ2 = q2.get128();
579 float32x4_t A0, A1, B1, A2, B2, A3, B3;
580 float32x2_t vQ1zx, vQ2wx, vQ1yz, vQ2zx, vQ2yz, vQ2xz;
581
582 {
583 float32x2x2_t tmp;
