43struct b3GpuConstraintInfo2
44{
45 // integrator parameters: frames per second (1/stepsize), default error
46 // reduction parameter (0..1).
47 b3Scalar fps,erp;
48
49 // for the first and second body, pointers to two (linear and angular)
50 // n*3 jacobian sub matrices, stored by rows. these matrices will have
51 // been initialized to 0 on entry. if the second body is zero then the
52 // J2xx pointers may be 0.
53 b3Scalar *m_J1linearAxis,*m_J1angularAxis,*m_J2linearAxis,*m_J2angularAxis;
54
55 // elements to jump from one row to the next in J's
56 int rowskip;
57
58 // right hand sides of the equation J*v = c + cfm * lambda. cfm is the
59 // "constraint force mixing" vector. c is set to zero on entry, cfm is
60 // set to a constant value (typically very small or zero) value on entry.
61 b3Scalar *m_constraintError,*cfm;
62
63 // lo and hi limits for variables (set to -/+ infinity on entry).
64 b3Scalar *m_lowerLimit,*m_upperLimit;
65
66 // findex vector for variables. see the LCP solver interface for a
67 // description of what this does. this is set to -1 on entry.
68 // note that the returned indexes are relative to the first index of
69 // the constraint.
70 int *findex;
71 // number of solver iterations
72 int m_numIterations;
73
74 //damping of the velocity
75 b3Scalar m_damping;
76};
