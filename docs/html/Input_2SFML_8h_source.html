100 }
101
102
103 case sf::Event::MouseButtonPressed:
104 case sf::Event::MouseButtonReleased:
105 {
106 return m_Canvas->InputMouseButton( event.MouseButton.Button, event.Type == sf::Event::MouseButtonPressed );
107 }
108
109 case sf::Event::MouseWheelMoved:
110 {
111 return m_Canvas->InputMouseWheel( event.MouseWheel.Delta * 60 );
112 }
113
114 case sf::Event::TextEntered:
115 {
116 return m_Canvas->InputCharacter( event.Text.Unicode );
117 }
118
119 case sf::Event::KeyPressed:
120 case sf::Event::KeyReleased:
121 {
122 bool bPressed = (event.Type == sf::Event::KeyPressed);
123
124 if ( event.Key.Control && bPressed && event.Key.Code >= 'a' && event.Key.Code <= 'z' )
125 {
126 return m_Canvas->InputCharacter( event.Key.Code );
127 }
128
129 unsigned char iKey = TranslateKeyCode( event.Key.Code );
130
131 return m_Canvas->InputKey( iKey, bPressed );
132
133 }
134 }
135
