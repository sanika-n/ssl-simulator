33static GLuint BindFont(const CTexFont *_Font)
34{

40 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);

43 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
44 glTexImage2D(GL_TEXTURE_2D, 0, GL_RED, _Font->m_TexWidth, _Font->m_TexHeight, 0, GL_RED, GL_UNSIGNED_BYTE, _Font->m_TexBytes);
45 glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
46 glTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
