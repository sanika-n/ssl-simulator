81/*
82 * Definition of macro used to set constants for size of network structures;
83 * machines with preprocessors that can't handle all of the sz_ symbols
84 * can define this macro to be sizeof(x) if and only if their compiler doesn't
85 * pad out structures (esp. the xTextElt structure which contains only two
86 * one-byte fields). Network structures should always define sz_symbols.
87 *
88 * The sz_ prefix is used instead of something more descriptive so that the
89 * symbols are no more than 32 characters long (which causes problems for some
90 * compilers and preprocessors).
91 *
92 * The extra indirection is to get macro arguments to expand correctly before
93 * the concatenation, rather than afterward.
94 */
95# define _SIZEOF(x) sz_##x
96# define SIZEOF(x) _SIZEOF(x)
97
