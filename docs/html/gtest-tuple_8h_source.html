942// k1 != k2.

951 }

956 template <class Tuple1, class Tuple2>

958 return SameSizeTuplePrefixComparator<k - 1, k - 1>::Eq(t1, t2) &&
959 ::std::tr1::get<k - 1>(t1) == ::std::tr1::get<k - 1>(t2);
960 }
