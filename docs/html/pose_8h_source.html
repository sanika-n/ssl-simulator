120 void getRPY(double &roll,double &pitch,double &yaw) const
121 {
122 double sqw;
123 double sqx;
124 double sqy;
125 double sqz;
126
127 sqx = this->x * this->x;
128 sqy = this->y * this->y;
129 sqz = this->z * this->z;
130 sqw = this->w * this->w;
131
132 roll = atan2(2 * (this->y*this->z + this->w*this->x), sqw - sqx - sqy + sqz);
133 double sarg = -2 * (this->x*this->z - this->w*this->y);
134 pitch = sarg <= -1.0 ? -0.5*M_PI : (sarg >= 1.0 ? 0.5*M_PI : asin(sarg));
135 yaw = atan2(2 * (this->x*this->y + this->w*this->z), sqw + sqx - sqy - sqz);

146 void setFromRPY(double roll, double pitch, double yaw)
147 {
148 double phi, the, psi;
149
150 phi = roll / 2.0;
151 the = pitch / 2.0;
152 psi = yaw / 2.0;
153
154 this->x = sin(phi) * cos(the) * cos(psi) - cos(phi) * sin(the) * sin(psi);
155 this->y = cos(phi) * sin(the) * cos(psi) + sin(phi) * cos(the) * sin(psi);
156 this->z = cos(phi) * cos(the) * sin(psi) - sin(phi) * sin(the) * cos(psi);
157 this->w = cos(phi) * cos(the) * cos(psi) + sin(phi) * sin(the) * sin(psi);
158
159 this->normalize();
160 };

175 {
176 double s = sqrt(this->x * this->x +
177 this->y * this->y +
178 this->z * this->z +
179 this->w * this->w);

182 this->x = 0.0;
183 this->y = 0.0;
184 this->z = 0.0;
185 this->w = 1.0;

197 Rotation operator*( const Rotation &qt ) const
198 {
199 Rotation c;
200
201 c.x = this->w * qt.x + this->x * qt.w + this->y * qt.z - this->z * qt.y;
202 c.y = this->w * qt.y - this->x * qt.z + this->y * qt.w + this->z * qt.x;
203 c.z = this->w * qt.z + this->x * qt.y - this->y * qt.x + this->z * qt.w;
204 c.w = this->w * qt.w - this->x * qt.x - this->y * qt.y - this->z * qt.z;

227 // Get the inverse of this quaternion
228 Rotation GetInverse() const
229 {
230 Rotation q;
231
232 double norm = this->w*this->w+this->x*this->x+this->y*this->y+this->z*this->z;
233
234 if (norm > 0.0)
235 {
236 q.w = this->w / norm;
237 q.x = -this->x / norm;
238 q.y = -this->y / norm;
239 q.z = -this->z / norm;
240 }
241
242 return q;
