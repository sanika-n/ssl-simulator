9 void b3PlaneSpace1 (b3Float4ConstArg n, b3Float4* p, b3Float4* q)
10{
11 if (b3Fabs(n.z) > 0.70710678f) {
12 // choose p in y-z plane
13 float a = n.y*n.y + n.z*n.z;
14 float k = 1.f/sqrt(a);
15 p[0].x = 0;
16 p[0].y = -n.z*k;
17 p[0].z = n.y*k;
18 // set q = n x p
19 q[0].x = a*k;
20 q[0].y = -n.x*p[0].z;
21 q[0].z = n.x*p[0].y;
22 }
23 else {
24 // choose p in x-y plane
25 float a = n.x*n.x + n.y*n.y;
26 float k = 1.f/sqrt(a);
27 p[0].x = -n.y*k;
28 p[0].y = n.x*k;
29 p[0].z = 0;
30 // set q = n x p
31 q[0].x = -n.z*p[0].y;
32 q[0].y = n.z*p[0].x;
33 q[0].z = a*k;
34 }
35}

55 float invMass0, const b3Mat3x3* invInertia0, float invMass1, const b3Mat3x3* invInertia1)
